8.1 Pendahuluan

Classical Test Theory (CTT) mengevaluasi kualitas soal melalui tiga indeks utama:

Indeks Simbol Rentang Kriteria Baik
Tingkat Kesulitan \(p\) 0 – 1 0.30 – 0.70
Daya Beda \(D\) / \(r_{pbis}\) −1 – +1 ≥ 0.30
Reliabilitas (Alpha) \(\alpha\) 0 – 1 ≥ 0.70

8.2 Data Simulasi

Kita gunakan data simulasi 30 peserta × 10 soal pilihan ganda (skor 0/1).

Code 
library(dplyr)
library(ggplot2)
library(knitr)
library(kableExtra)

set.seed(42)

n_peserta <- 30
n_soal    <- 10

# Simulasi kemampuan peserta (bervariasi)
theta <- rnorm(n_peserta, mean = 0, sd = 1)

# Tingkat kesulitan soal (bervariasi sengaja ada yg mudah, sedang, sulit)
p_true <- c(0.85, 0.75, 0.65, 0.55, 0.50, 0.45, 0.35, 0.25, 0.15, 0.10)

# Generate respon (soal lebih mudah = peluang benar lebih tinggi)
data_soal <- sapply(p_true, function(p) {
  # Simulasi dengan variasi kemampuan peserta
  p_adj <- plogis(qlogis(p) + theta * 0.8)
  rbinom(n_peserta, 1, p_adj)
})

colnames(data_soal) <- paste0("Soal_", sprintf("%02d", 1:n_soal))
rownames(data_soal) <- paste0("Peserta_", sprintf("%02d", 1:n_peserta))

# Tampilkan 6 baris pertama
kable(data_soal[1:6, ],
      caption = "Contoh Data Respon (6 peserta pertama)",
      align = "c") |>
  kable_styling(bootstrap_options = c("striped", "hover"),
                full_width = FALSE)

8.5 Reliabilitas

8.5.1 Cronbach Alpha

\[\alpha = \frac{k}{k-1} \left(1 - \frac{\sum \sigma^2_i}{\sigma^2_X}\right)\]

Code 
# Hitung alpha secara manual
k          <- n_soal
var_item   <- apply(data_soal, 2, var)
var_total  <- var(skor_total)
alpha      <- (k / (k - 1)) * (1 - sum(var_item) / var_total)

# Alpha jika item dihapus
alpha_jika_hapus <- sapply(1:k, function(i) {
  skor_tanpa_i <- rowSums(data_soal[, -i])
  var_item_i   <- apply(data_soal[, -i], 2, var)
  var_tot_i    <- var(skor_tanpa_i)
  ((k-1) / (k-2)) * (1 - sum(var_item_i) / var_tot_i)
})

cat(sprintf("Cronbach Alpha  : %.3f\n", alpha))
Cronbach Alpha  : 0.700
Code 
cat(sprintf("Interpretasi    : %s\n",
    ifelse(alpha >= 0.90, "Sangat Baik (>= .90)",
    ifelse(alpha >= 0.80, "Baik (.80–.89)",
    ifelse(alpha >= 0.70, "Cukup (.70–.79)",
           "Perlu perbaikan (< .70)")))))
Interpretasi    : Cukup (.70–.79)

df_alpha <- data.frame(

## Reliabilitas

### Cronbach Alpha

$$\alpha = \frac{k}{k-1} \left(1 - \frac{\sum \sigma^2_i}{\sigma^2_X}\right)$$

```{r}
#| label: reliabilitas

# Hitung alpha secara manual
k          <- n_soal
var_item   <- apply(data_soal, 2, var)
var_total  <- var(skor_total)
alpha      <- (k / (k - 1)) * (1 - sum(var_item) / var_total)

# Alpha jika item dihapus
alpha_jika_hapus <- sapply(1:k, function(i) {
  skor_tanpa_i <- rowSums(data_soal[, -i])
  var_item_i   <- apply(data_soal[, -i], 2, var)
  var_tot_i    <- var(skor_tanpa_i)
  ((k-1) / (k-2)) * (1 - sum(var_item_i) / var_tot_i)
})

cat(sprintf("Cronbach Alpha  : %.3f\n", alpha))
cat(sprintf("Interpretasi    : %s\n",
    ifelse(alpha >= 0.90, "Sangat Baik (>= .90)",
    ifelse(alpha >= 0.80, "Baik (.80–.89)",
    ifelse(alpha >= 0.70, "Cukup (.70–.79)",
           "Perlu perbaikan (< .70)")))))

df_alpha <- data.frame(
  Soal              = colnames(data_soal),
  Alpha_jika_hapus  = round(alpha_jika_hapus, 3),
  Sinyal            = ifelse(alpha_jika_hapus > alpha,
                             "Naik jika dihapus", "Turun jika dihapus")
)

kable(df_alpha,
      col.names = c("Soal", "Alpha Jika Dihapus", "Sinyal"),
      align = "c",
      caption = sprintf("Alpha Cronbach = %.3f | Alpha Jika Item Dihapus", alpha)) |>
  kable_styling(bootstrap_options = c("striped", "hover"),
                full_width = FALSE) |>
  row_spec(which(df_alpha$Sinyal == "Naik jika dihapus"),
           background = "#fdecea")
```
